450void KMPSearch(const char *pattern, const char *text) {
451 int m = strlen(pattern);
452 int n = strlen(text);
453 int *lps = (int *)malloc(m * sizeof(int)); // Allocate memory for LPS array
454 int j = 0;
455 // Compute LPS array for the pattern
456 lps[0] = 0;
457
458 for (int i = 1; i < m;) {
459 if (pattern[i] == pattern[j]) {
460 lps[i++] = ++j;
461 } else if (j) {
462 j = lps[j - 1];
463 } else {
464 lps[i++] = 0;
465 }
466 }
467
468 // Search for the pattern in the text
469 for (int i = 0; i < n;) {
470 if (pattern[j] == text[i]) {
471 ++j;
472 ++i;
473 }
474
475 if (j == m) {
476 printf("Pattern found at index %d\n", i - j);
477 j = lps[j - 1];
478 } else if (i < n && pattern[j] != text[i]) {
479 j = j ? lps[j - 1] : 0;
480 }
481 }
482
483 free(lps); // Free the dynamically allocated memory
484}

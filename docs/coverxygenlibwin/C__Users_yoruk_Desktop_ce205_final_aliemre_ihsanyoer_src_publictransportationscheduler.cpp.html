 798 * @param isLeaf Indicates if the node is a leaf node.
 799 * @return Pointer to the newly created B+ tree node.
 800 */
 1801BPlusTreeNode *createNode(bool isLeaf) {
 802  BPlusTreeNode *newNode = (BPlusTreeNode *)malloc(sizeof(BPlusTreeNode));
 803  newNode->isLeaf = isLeaf;
 804  newNode->numKeys = 0;
 805  newNode->next = NULL;
 806
 807  for (int i = 0; i <= MAX_KEYS; i++) {
 808    newNode->children[i] = NULL;
 809  }
 810
 811  return newNode;

 814/**
 815 * @brief Searches the B+ tree for a specific key.
 816 * @param key The key to search for.
 817 * @return Pointer to the ticket sale if found, NULL otherwise.
 818 */
 1819TicketSale *searchBPlusTree(int key) {
 820  BPlusTreeNode *current = root;
 821
 822  while (current != NULL) {
 823    int i = 0;
 824
 825    while (i < current->numKeys && key > current->keys[i]) {
 826      i++;
 827    }
 828
 829    if (i < current->numKeys && key == current->keys[i]) {
 830      return current->sales[i];
 831    }
 832
 833    if (current->isLeaf) {
 834      return NULL;
 835    }
 836
 837    current = current->children[i];
 838  }
 839
 840  return NULL;
 841}

 843/**
 844 * @brief Inserts a new key and ticket sale into the B+ tree.
 845 * @param key The key to insert.
 846 * @param sale The associated ticket sale data.
 847 */
 1848void insertBPlusNode(int key, TicketSale *sale) {
 849  if (root == NULL) {
 850    root = createNode(true);
 851    root->keys[0] = key;
 852    root->sales[0] = sale;
 853    root->numKeys = 1;
 854    return;
 855  }
 856
 857  BPlusTreeNode *current = root;
 858  BPlusTreeNode *parent = NULL;
 859
 860  while (!current->isLeaf) {
 861    parent = current;
 862    int i = 0;
 863
 864    while (i < current->numKeys && key > current->keys[i]) {
 865      i++;
 866    }
 867
 868    current = current->children[i];
 869  }
 870
 871  if (current->numKeys < MAX_KEYS) {
 872    int i = current->numKeys - 1;
 873
 874    while (i >= 0 && current->keys[i] > key) {
 875      current->keys[i + 1] = current->keys[i];
 876      current->sales[i + 1] = current->sales[i];
 877      i--;
 878    }
 879
 880    current->keys[i + 1] = key;
 881    current->sales[i + 1] = sale;
 882    current->numKeys++;
 883  }
 884}
